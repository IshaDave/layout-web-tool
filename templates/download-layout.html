Don't forget to sign the downloaded layout with you project owner private key. 
>>> from in_toto.models.layout import Layout
>>> from in_toto.util import import_rsa_key_from_file
>>> layout = Layout.read_from_file("root.layout")
>>> alice_private = import_rsa_key_from_file("alice", password="123456")
>>> layout.sign(alice_private)
>>> layout.dump("roo.layout")
